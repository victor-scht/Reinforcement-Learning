			\begin{block}{Motivation}

				\begin{itemize}
					\item Reinforcement Learning (RL) agents optimize the \textbf{reward function}, not the true objective.

					      \medskip

					\item In practice, reward functions are often \textbf{imperfect proxies} of the desired behavior.

					      \medskip

					\item This mismatch can lead to:

					      \medskip

					      \begin{itemize}
						      \item unintended strategies

						            \medskip

						      \item suboptimal or unsafe behaviors

						            \medskip

						      \item complete task failure despite high reward
					      \end{itemize}
				\end{itemize}

				\medskip

				\begin{alertblock}{Key issue}


					\centerline{\textit{``Agents do exactly what you reward, not what you intend.''}}

					\medskip

					This phenomenon is known as \textbf{reward misspecification} or \textbf{reward hacking} (Amodei et al., 2016).

				\end{alertblock}


			\end{block}

			% ---------------------------------------------------------

			\begin{block}{Environment}

				\begin{itemize}
					\item Gridworld with obstacles, cliffs, risky zones
					\item Multiple possible paths (no trivial solution)
					\item \textbf{Stochastic transitions}:

					      \medskip

					      \begin{itemize}
						      \item action noise (slip probability)

						            \medskip

						      \item intended action not always executed
					      \end{itemize}
				\end{itemize}

				\medskip

				\textbf{Impact:}
				\begin{itemize}
					\item harder exploration
					\item increases importance of reward signal
				\end{itemize}

			\end{block}

			% ---------------------------------------------------------

			\begin{block}{Reward Design and Theory}

				\textbf{Correct reward}
				\[
					R(s,a)=
					\begin{cases}
						+10 & \text{goal}  \\
						-1  & \text{step}  \\
						-50 & \text{cliff}
					\end{cases}
				\]

				\medskip

				\textbf{Misspecified reward (reward hacking)}
				\[
					R'(s,a)=R(s,a)+\beta \cdot (-d(s,G))
				\]

				\begin{itemize}
					\item dense but biased signal
					\item encourages proximity instead of success
				\end{itemize}

				\medskip

				\textbf{Potential-based shaping (Ng et al., 1999)}
				\[
					R_{\text{pot}}(s,a,s')=R(s,a)+\gamma \Phi(s')-\Phi(s)
				\]

				\begin{itemize}
					\item $\Phi(s)=-d(s,G)$
					\item \textbf{guarantee: optimal policy unchanged}
					\item improves learning speed
				\end{itemize}

				\begin{alertblock}{Learning Dynamics (Intuition)}

					\begin{itemize}
						\item Correct reward: sparse $\rightarrow$ slow but correct
						\item Proximity reward: dense $\rightarrow$ fast but wrong
						\item Potential shaping:
						      \begin{itemize}
							      \item dense signal

							            \medskip

							      \item preserves optimal policy

							            \medskip

							      \item \textbf{best trade-off}
						      \end{itemize}
					\end{itemize}

				\end{alertblock}

			\end{block}

			% ---------------------------------------------------------

			\begin{block}{Learning Algorithms}

				\textbf{Q-learning (off-policy)}
				\[
					Q \leftarrow r + \gamma \max_a Q
				\]

				\textbf{SARSA (on-policy)}
				\[
					Q \leftarrow r + \gamma Q(a')
				\]

				\medskip

				\textbf{Key difference}

				\begin{itemize}
					\item Q-learning assumes optimal future
					\item SARSA learns under exploration policy
				\end{itemize}

				\medskip

				% \textbf{Why SARSA is slower}
				%
				% \begin{itemize}
				% 	\item incorporates noisy actions (due to $\epsilon$ + environment noise)
				% 	\item propagates conservative values
				% 	\item safer but slower convergence
				% \end{itemize}

			\end{block}

					\item Misspecified reward $\rightarrow$ loops / local exploitation
					\item Correct $\rightarrow$ cautious but suboptimal paths
					\item Potential $\rightarrow$ efficient shortest paths
				\end{itemize}

			\end{block}
			% ---------------------------------------------------------

			\begin{alertblock}{Main Insight}

				\begin{center}
					\Large
					Reward design dominates learning behavior \\
					\medskip
					High return $\neq$ task success
				\end{center}

			\end{alertblock}

			% ---------------------------------------------------------

			\begin{block}{Conclusion}

				\begin{itemize}

					\item \textbf{Reward design strongly impacts learning}: agents optimize the signal they receive, not the intended objective.

					\item \textbf{Misspecified rewards lead to reward hacking}, producing high returns but incorrect behaviors.

					\item With stochastic transitions:

					      \medskip

					      \begin{itemize}
						      \item \textbf{Q-learning} learns faster but is less robust

						            \medskip

						      \item \textbf{SARSA} is more stable but slower
					      \end{itemize}

					\item \textbf{Potential-based shaping} improves learning speed while preserving the optimal policy.

				\end{itemize}

				\medskip

				\begin{center}
					\textbf{Designing the reward is as important as choosing the algorithm.}
				\end{center}

			\end{block}
